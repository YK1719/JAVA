public class imp19 { 
    public static void main(String[] args) { 
        int num = Integer.parseInt(args[0]); // Read number from command line 
 
        System.out.print("Factors of " + num + " are: "); 
 
        for (int i = 1; i <= num; i++) { 
            if (num % i == 0) { 
                System.out.print(i + " "); 
            } 
        } 
        System.out.println(); 
    } 
}
