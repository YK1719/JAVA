#include <stdio.h> 
#include <unistd.h> #include <sys/types.h> int main() {     pid_t pid = fork();    if (pid == 0) {          nice(-5);  // Increase priority         printf("Child process running with higher priority, PID = %d\n", getpid()); 
    } else if (pid > 0) {  
        printf("Parent process running, PID = %d\n", getpid()); 
    } else {         printf("Fork failed\n");  }     return 0;}
