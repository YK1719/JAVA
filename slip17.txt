        }

    public void isEven()

class Student { 
    int rollno; 
    String name; 
    static int count = 0; // static variable to keep object count 
 
    // Default constructor 
    Student() { 
        count++; 
    } 
 
    // Parameterized constructor 
    Student(int rollno, String name) { 
        this.rollno = rollno; 
        this.name = name; 
        count++; 
        System.out.println("Object created: " + name + " (Roll No: " + rollno + ")"); 
        System.out.println("Total objects = " + count); 
        System.out.println(); 
    } 
} 
 
public class imp18 { 
    public static void main(String[] args) { 
        Student s1 = new Student(1, "Amit"); 
        Student s2 = new Student(2, "Sneha"); 
        Student s3 = new Student(3, "Raj"); 
    } 
}
