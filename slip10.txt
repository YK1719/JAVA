import pandas as pd
from apyori import apriori

# Load dataset
data = pd.read_csv("Iris.csv")

# Prepare transactions (convert each row to a list of strings)
records = []
for i in range(len(data)):
    records.append([str(data.iloc[i, j]) for j in range(len(data.columns))])

# Apply Apriori algorithm
association_rules = apriori(records, 
                            min_support=0.0045, 
                            min_confidence=0.2, 
                            min_lift=3, 
                            min_length=2)

# Convert results to list
results = list(association_rules)

# Print results
for item in results:
    print(item)

# Print total number of rules
print("Total association rules:", len(results))
