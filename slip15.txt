import java.util.*; 
 
public class imp16 { 
    public static void main(String[] args) { 
        // Create LinkedList 
        LinkedList<String> fruits = new LinkedList<>(); 
 
        // Add fruits 
        fruits.add("Apple"); 
        fruits.add("Banana"); 
        fruits.add("Guava"); 
        fruits.add("Orange"); 
 
        // Display in reverse using ListIterator 
        System.out.println("Fruits in reverse order:"); 
        ListIterator<String> itr = fruits.listIterator(fruits.size()); 
        while (itr.hasPrevious()) { 
            System.out.println(itr.previous()); 
        } 
    } 
}
