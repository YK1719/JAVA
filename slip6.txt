import java.io.*; 
import java.util.*; 
class emp 
{ 
    int id,sal; 
    String name; 
    emp() 
    { 
        id=1; 
        sal=9000; 
        name="nilima"; 
    } 
    public void accept() 
    { 
        Scanner sc=new Scanner(System.in); 
        System.out.println("enter the emp id"); 
        id=sc.nextInt(); 
        System.out.println("enter the emp salary"); 
        sal=sc.nextInt(); 
        System.out.println("enter the emp name"); 
        name=sc.next(); 
    } 
  public void display() 
    { 
        System.out.println("the emp id is:"+id); 
        System.out.println("the emp sal is:"+sal); 
        System.out.println("the emp name is:"+name); 
    } 
} 
class manager extends emp 
{ 
    private int bonus; 
    manager() 
    { 
        bonus=10; 
    } 
      public void accept() 
    { 
        super.accept(); 
        Scanner s1=new Scanner(System.in); 
        System.out.println("enter the emp bonus"); 
        bonus=s1.nextInt(); 
    } 
    public void display() 
    { 
        super.display(); 
        System.out.println("the emp bonus "+bonus); 
        System.out.println("Total Salary (Salary+Bonus):"+(sal+bonus)); 
    } 
 public int sal() 
    { 
        return sal+bonus; 
    } 
}
