# Create the DataFrame
students <- data.frame(
  Rollno = c(1, 2, 3, 4, 5),
  Studname = c("Alice", "Bob", "Charlie", "David", "Eva"),
  Address = c("Mumbai", "Pune", "Nagpur", "Nashik", "Aurangabad"),
  Marks = c(85, 90, 78, 88, 92)

print(students)






# Import necessary libraries
import pandas as pd
from sklearn.linear_model import LinearRegression

# Load dataset
url = "https://www.w3schools.com/python/ML_data.csv"  # CSV file from W3Schools
data = pd.read_csv(url)

# Define features and target
X = data[['Weight', 'Volume']]  # Features
y = data['CO2']                 # Target

# Create and train the model
model = LinearRegression()
model.fit(X, y)

# Print estimated coefficients
print("Intercept (b0):", model.intercept_)
print("Coefficients (b1, b2):", model.coef_)

# Make predictions (optional)
y_pred = model.predict(X)
print("\nPredicted CO2 values:", y_pred)
